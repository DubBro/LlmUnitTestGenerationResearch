        [Fact]
        public void ToCsvCompatible_StringIsEmpty_ReturnsEmptyString()
        {
            // Arrange
            string s = "";

            // Act
            var result = s.ToCsvCompatible();

            // Assert
            Assert.Equal("", result);
        }

        [Theory]
        [InlineData("simple")]
        [InlineData("with space")]
        [InlineData("with,comma")]
        [InlineData("with;semicolon")]
        [InlineData("with\nnewline")]
        public void ToCsvCompatible_StringContainsSpecialChars_ReturnsQuotedString(string input)
        {
            // Arrange
            string s = input;

            // Act
            var result = s.ToCsvCompatible();

            // Assert
            Assert.StartsWith("\"", result);
            Assert.EndsWith("\"", result);
        }

        [Fact]
        public void ToCsvCompatible_StringContainsDoubleQuotes_ReturnsEscapedString()
        {
            // Arrange
            string s = "with\"quote";

            // Act
            var result = s.ToCsvCompatible();

            // Assert
            Assert.Equal("\"with\"\"quote\"", result);
        }

        [Fact]
        public void ToCsvCompatible_StringStartsWithSpace_ReturnsQuotedString()
        {
            // Arrange
            string s = " start";

            // Act
            var result = s.ToCsvCompatible();

            // Assert
            Assert.Equal("\" start\"", result);
        }

        [Fact]
        public void ToCsvCompatible_StringEndsWithSpace_ReturnsQuotedString()
        {
            // Arrange
            string s = "end ";

            // Act
            var result = s.ToCsvCompatible();

            // Assert
            Assert.Equal("\"end \"", result);
        }
