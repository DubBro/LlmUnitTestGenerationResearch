using Dataset.Sample13;
using NSubstitute;

namespace Glm45AirUnitTests;

public class OntologyHelperTests
{
    [Fact]
    public void ParseNode_NodeTypeIsUri_ReturnsParsedString()
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Uri);
        node.ToString().Returns("http://example.org#123");
        var expected = "123";

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }

    [Fact]
    public void ParseNode_NodeTypeIsLiteral_ReturnsParsedString()
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Literal);
        node.ToString().Returns("123^http://www.w3.org/2001/XMLSchema#string");
        var expected = "123";

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }

    [Fact]
    public void ParseNode_NodeTypeIsBlank_ReturnsEmptyString()

    [Fact]
    public void ParseNode_NodeToStringReturnsNull_ReturnsEmptyString()
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Uri);
        node.ToString().Returns((string)null);
        var expected = string.Empty;

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }

    [Theory]
    [InlineData("", "")]
    [InlineData("http://example.org#", "")]
    [InlineData("http://example.org#123", "123")]
    [InlineData("http://example.org#123#456", "123#456")]
    public void ParseNode_NodeTypeIsUri_WithVariousUris_ReturnsCorrectSubstring(string input, string expected)
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Uri);
        node.ToString().Returns(input);

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }

    [Theory]
    [InlineData("", "")]
    [InlineData("123^", "123")]
    [InlineData("123^http://www.w3.org/2001/XMLSchema#string", "123")]
    [InlineData("123^http://www.w3.org/2001/XMLSchema#string^extra", "123")]
    public void ParseNode_NodeTypeIsLiteral_WithVariousLiterals_ReturnsCorrectSubstring(string input, string expected)
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Literal);
        node.ToString().Returns(input);

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }

    [Fact]
    public void ParseNode_NodeTypeIsUri_WithEncodedCharacters_ReturnsDecodedString()
    {
        // Arrange
        var node = Substitute.For<INode>();
        node.NodeType.Returns(NodeType.Uri);
        node.ToString().Returns("http://example.org/value%23with%20hash");
        var expected = "value#with hash";

        // Act
        var actual = OntologyHelper.ParseNode(node);

        // Assert
        Assert.Equal(expected, actual);
    }
